25  	protected Movement[] getMoveOrder(final int x, final int y) {
26  		Movement[] moves = new Movement[4];
27  		Direction first, second;
28  		// If it's above:
29  		if (y < 0) {
30  			// If it's to the left:
31  			if (x < 0) {
32  				first = Direction.UL;
33  				second = (x < y) ? Direction.DL : Direction.UR;
34  			}
35  			// If it's to the right:
36  			else {
37  				first = Direction.UR;
38  				second = (-x < y) ? Direction.DR : Direction.UL;
39  			}
40  		}
41  		// If it's below:
42  		else {
43  			// If it's to the left:
44  			if (x < 0) {
45  				first = Direction.DL;
46  				second = (x < -y) ? Direction.UL : Direction.DR;
47  			}
48  			// If it's to the right:
49  			else {
50  				first = Direction.DR;
51  				second = (x > y) ? Direction.UR : Direction.DL;
52  			}
53  		}
54  		moves[0] = new Movement(first);
55  		moves[1] = new Movement(second);
56  		moves[2] = new Movement(second);
57  		moves[2].reverse();
58  		moves[3] = new Movement(first);
59  		moves[3].reverse();
60  		return moves;
61  	}
