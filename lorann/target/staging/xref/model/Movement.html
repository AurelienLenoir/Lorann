15  //----------------------ATTRIBUTES--------------------
16  	/**
17  	 * Associates a Direction with a Point.
18  	 */
19  	private final static HashMap<Direction, Point> MAP = new HashMap<>();
20  	private final static ArrayList<Direction> LIST = new ArrayList<>();
21  	private final static HashMap<Double, Direction> ANGLE = new HashMap<>();
22  	
23  	static {
24  		MAP.put(Direction.U, new Point(0, -1));
25  		MAP.put(Direction.D, new Point(0, 1));
26  		MAP.put(Direction.L, new Point(-1, 0));
27  		MAP.put(Direction.R, new Point(1, 0));
28  		MAP.put(Direction.UL, new Point(-1, -1));
29  		MAP.put(Direction.UR, new Point(1, -1));
30  		MAP.put(Direction.DL, new Point(-1, 1));
31  		MAP.put(Direction.DR, new Point(1, 1));
32  		MAP.put(Direction.NONE, new Point(0, 0));
33  		
34  		LIST.add(Direction.U);
35  		LIST.add(Direction.UR);
36  		LIST.add(Direction.R);
37  		LIST.add(Direction.DR);
38  		LIST.add(Direction.D);
39  		LIST.add(Direction.DL);
40  		LIST.add(Direction.L);
41  		LIST.add(Direction.UL);
42  		
43  		ANGLE.put(0.0,			Direction.R);
44  		ANGLE.put(1.0/4.0,		Direction.UR);
45  		ANGLE.put(1.0/2.0,		Direction.U);
46  		ANGLE.put(3.0/4.0,		Direction.UL);
47  		ANGLE.put(1.0,			Direction.L);
48  		ANGLE.put(-1.0,			Direction.L);  // Don't forget -pi. Never.
49  		ANGLE.put(-3.0/4.0,		Direction.DL);
50  		ANGLE.put(-1.0/2.0,		Direction.D);
51  		ANGLE.put(-1.0/4.0,		Direction.DR);
52  	}
53  	
54  	private Direction dir;
55  	
56  //----------------------METHODS---------------------	
57  	public Movement(Direction dir) {
58  		this.setDirection(dir);
59  	}
60  	
61  	public Movement(Point vec) {
62  		this.setDirection(vec);
63  	}
64  	
65  	public int getX() {
66  		return (int) MAP.get(dir).getX();
67  	}
68  	
69  	public int getY() {
70  		return (int) MAP.get(dir).getY();
71  	}
72  	
73  	public void rotateLeft() {
74  		Direction dir;
75  		int i = LIST.indexOf(this.dir)-1;
76  		if (i < 0) {
77  			i = LIST.size()-1;
78  		}
79  		dir = LIST.get(i);
80  		this.setDirection(dir);
81  	}
82  	
83  	public void rotateRight() {
84  		Direction dir;
85  		int i = LIST.indexOf(this.dir)+1;
86  		if (i > 7) {
87  			i = 0;
88  		}
89  		dir = LIST.get(i);
90  		this.setDirection(dir);		
91  	}
92  	
93  	public void reverse() {
94  		Direction dir;
95  		int i = LIST.indexOf(this.dir)+4;
96  		if (i > 7) {
97  			i -= 8;
98  		}
99  		dir = LIST.get(i);
100 		this.setDirection(dir);
101 		
102 	}
103 	
104 	public void setDirection(Direction dir) {
105 		this.dir = dir;
106 	}
107 	
108 	public void setDirection(Point vec) {
109 		if (vec.x == 0.0 && vec.y == 0.0) {
110 			this.setDirection(Direction.NONE);
111 		}
112 		else {
113 			double a = Math.atan2(-vec.getY(), vec.getX()) / Math.PI;
114 			double d = 1.0/8.0;
115 			Direction dir;
116 			for (final double angle : ANGLE.keySet()) {
117 				dir = ANGLE.get(angle);
118 				if ( a >= (angle-d) && a <= (angle+d) ) {
119 					this.setDirection(dir);
120 					return;
121 				}
122 			}
123 		}
124 	}
125 	
126 	public Direction getDirection() {
127 		return this.dir;
128 	}
